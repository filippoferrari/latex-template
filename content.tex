\begin{algorithm}
\algsetup{indent=2em}
\caption{\texttt{example_algo}$(arg1, arg2)$ - Example of an algorithm}
\label{alg:createDecisionTree}
\begin{algorithmic}
\REQUIRE{the subsample of the dataset and the depth.}
\ENSURE{A decision tree.}
\STATE{}
\IF[BASE CASE]{$depth = 0$ \OR all examples have same label}
    \RETURN{most common label in the subsample}
\ENDIF
\STATE{}
\FOR[RECURSIVE CASE]{each feature}
    \STATE{Try splitting the data (i.e. build a decision stump)}
    \STATE{Calculate the cost/gain for this stump}
\ENDFOR
\STATE{Pick feature with maximum information gain/minimum cost}
\STATE{}
\STATE{Find left/right subsamples}
\STATE{Add left branch $\leftarrow$ \texttt{BuildTree}$(leftSubSample, depth - 1)$}
\STATE{Add right branch $\leftarrow$ \texttt{BuildTree}$(rightSubSample, depth - 1)$}
\RETURN{$tree$}
\end{algorithmic}
\end{algorithm}
